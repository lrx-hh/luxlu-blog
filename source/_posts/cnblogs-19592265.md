import sys
import re

def is_base64(s):
    """判断字符串是否可能是 Base64 编码"""
    if isinstance(s, str):
        # Base64 只包含字母、数字、+、/，结尾最多两个 '='
        s = s.strip()
        if len(s) == 0:
            return False
        if not re.fullmatch(r'[A-Za-z0-9+/]*={0,2}', s):
            return False
        # 长度必须是 4 的倍数
        if len(s) % 4 != 0:
            return False
        return True
    return False

def multi_decode(data):
    """多次 Base64 解码，直到失败或得到非 Base64 内容"""
    current = data.encode() if isinstance(data, str) else data
    step = 0

    while True:
        try:
            # 尝试 Base64 解码
            decoded = base64.b64decode(current)
            step += 1
            print(f"[Step {step}] Decoded: {decoded}")

            # 检查解码后的内容是否仍是 Base64 字符串（作为文本）
            try:
                decoded_str = decoded.decode('utf-8')
                if is_base64(decoded_str):
                    current = decoded  # 下一轮用 bytes 继续解码
                    continue
                else:
                    print("\n[+] Final result (not Base64 anymore):")
                    print(decoded_str)
                    return decoded_str
            except UnicodeDecodeError:
                # 如果不是文本（比如是二进制），则停止
                print("\n[+] Final result is binary data:")
                print(decoded)
                return decoded

        except Exception as e:
            print(f"\n[!] Decoding stopped at step {step}: {e}")
            try:
                print("Last readable content:", current.decode('utf-8', errors='replace'))
            except:
                print("Last content (binary):", current)
            return current

if __name__ == "__main__":
    if len(sys.argv) &lt; 2:
        print("Usage:")
        print("  python multi_base64_decode.py &lt;string&gt;")
        print("  python multi_base64_decode.py file:&lt;filename&gt;")
        sys.exit(1)

    input_arg = sys.argv[1]

    if input_arg.startswith("file:"):
        filename = input_arg[5:]
        with open(filename, 'r') as f:
            content = f.read().strip()
    else:
        content = input_arg.strip()

    print("[*] Starting multi-layer Base64 decoding...")
    print("[Input]:", content)
    print("-" * 50)
    multi_decode(content)

<pre><code class="language-python">import numpy as np
from scipy.io import wavfile
 
def extract_binary_from_wav(file_path, threshold_freq, output_path):
    sample_rate, data = wavfile.read(file_path)
    
    if len(data.shape) &gt; 1:
        data = data[:, 0]
    
    window_size = int(sample_rate * 0.01)  # 10ms窗口
    binary_data = []
 
    for i in range(0, len(data), window_size):
        window = data[i:i + window_size]
        if len(window) &lt; window_size:
            break
 
        fft_result = np.fft.fft(window)
        freqs = np.fft.fftfreq(len(window), d=1/sample_rate)
        
        magnitude = np.abs(fft_result)
        
        max_index = np.argmax(magnitude[:len(magnitude)//2])
        dominant_freq = freqs[max_index]
 
        binary_value = 1 if dominant_freq &gt;= threshold_freq else 0
        binary_data.append(binary_value)
 
    with open(output_path, "w") as f:
        binary_string = "".join(map(str, binary_data))
        f.write(binary_string)
    print(f"提取的二进制数据已保存到 {output_path}")
 
file_path = "attachment.wav"
output_path = "output.txt"

<pre><code class="language-python"># -*- coding: utf-8 -*-
# 差值音频生成脚本：left.raw + right.raw → all.raw

LEFT_FILE = 'l'      # 左声道 RAW 文件（16-bit, little-endian, 单声道）
RIGHT_FILE = 'r'     # 右声道 RAW 文件
OUTPUT_FILE = 'all'  # 输出的差值文件

# 假设原始立体声音频总大小为 31163904 字节（含左右声道）
# 每帧 = 左(2B) + 右(2B) = 4 字节 → 总帧数 = 31163904 // 4
TOTAL_BYTES = 31163904
FRAME_COUNT = TOTAL_BYTES // 4

def main():
    with open(LEFT_FILE, 'rb') as fl, open(RIGHT_FILE, 'rb') as fr:
        data = []
        for _ in range(FRAME_COUNT):
            # 读取一帧（4字节），但只取有效部分
            left_bytes = fl.read(2)
            right_bytes = fr.read(2)

            # 如果读到末尾，提前退出
            if len(left_bytes) &lt; 2 or len(right_bytes) &lt; 2:
                break

            # 将小端序字节转为有符号16位整数
            numl = int.from_bytes(left_bytes, byteorder='little', signed=True)
            numr = int.from_bytes(right_bytes, byteorder='little', signed=True)

            # 计算绝对差值（你也可以用 numl - numr 看正负）
            diff = abs(numl - numr)
            # 限制在 16 位无符号范围 [0, 65535]
            diff = min(diff, 65535)
            data.append(diff)

    # 写入输出文件（16-bit, little-endian, 无符号）
    with open(OUTPUT_FILE, 'wb') as f:
        for val in data:
            f.write(val.to_bytes(2, byteorder='little', signed=False))

    print(f"[+] 成功生成 '{OUTPUT_FILE}'，共 {len(data)} 个采样点")
    print("[+] 请在 Audacity 中导入为：")
    print("    - 编码：Signed 16-bit PCM（或 Unsigned 也可）")
    print("    - 字节序：小尾端 (Little-endian)")
    print("    - 声道：1 声道（单声道）")
    print("    - 采样率：44100 Hz")

if __name__ == '__main__':
    main()

<h1 id="sstv">SSTV</h1>
<p>慢扫描电视<code>Slow Scan Television</code></p>
<p>听起来是短促的电音</p>
<p>这是一种用无线电传输图像的方法，特别用于业余无线电爱好者之间发送图片。</p>
<p>它不是用来发照片的普通相机，而是将图像逐行转换成声音信号，然后通过无线电传播，接收方再把这些声音还原成图像。</p>
<ul>
<li>图像被分成很多水平线。</li>
<li>每一条线会被翻译成一段连续变化的音频信号。</li>
<li>不同的颜色和亮度对应不同的频率。</li>
<li>接收端播放这段音频，软件会根据频率的变化重新画出图像。</li>
</ul>
<p>如果你拿到一个 <code>&lt;font style="background-color:rgba(175, 184, 193, 0.2);"&gt;.wav&lt;/font&gt;</code> 文件，但听上去只是嗡嗡的噪音，它可能其实是一张图</p>
<p><font style="color: rgba(17, 17, 51, 0.5)"></font></p>
