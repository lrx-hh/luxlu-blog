def modinv(a, m):

        if (a * x) % m == 1:

def decrypt(ciphertext, a, b):
    a_inv = modinv(a, 26)
    if a_inv is None:

    plaintext = ""

            c = ord(char) - ord('a')

            plaintext += chr(m + ord('a'))
